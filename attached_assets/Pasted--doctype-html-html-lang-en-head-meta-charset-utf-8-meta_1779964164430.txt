const stageOrder = ["compose", "enhance", "clarify", "run", "report", "enrich"];

function gotoStageById(s) {
  if (!reached.includes(s)) reached.push(s);
  stageOrder.forEach(x => document.getElementById("stage-" + x).classList.toggle("hide", x !== s));
  document.querySelectorAll("#stage-bar .stage").forEach(el => {
    const x = el.dataset.stage; el.classList.remove("active", "done");
    if (x === s) el.classList.add("active");
    else if (reached.includes(x) && stageOrder.indexOf(x) < stageOrder.indexOf(s)) el.classList.add("done");
  });
  if (s === "enhance") buildEnhanced();
  if (s === "report") renderReport();
  window.scrollTo({ top: 0, behavior: "smooth" });
  log("nav", "Stage → " + s);
}
function gotoStage(el) { gotoStageById(el.dataset.stage); }
function gotoLayer(n) {
  for (let i = 1; i <= 4; i++) document.getElementById("layer-" + i).classList.toggle("hide", i !== n);
  document.querySelectorAll("#crumbs .crumb").forEach((c, i) => {
    c.classList.remove("active", "done");
    if (i + 1 < n) c.classList.add("done");
    if (i + 1 === n) c.classList.add("active");
  });
  if (n === 4) renderAskFilters();
  window.scrollTo({ top: 0, behavior: "smooth" });
}

function doExport(fmt) {
  const data = {
    shape: state.reportShape, target: state.target, industry: state.industry,
    countries: Object.keys(state.countries).filter(k => state.countries[k]),
    leads: [{ name: "Mohammed AlQasem", title: "CTO", company: "Tabby" }, { name: "Hisham Al-Falih", title: "CTO", company: "Lean Tech" }],
    companies: [{ name: "Tabby", funding: "$50M", employees: "250-500" }],
    sources: sourceReco(), generatedAt: new Date().toISOString(),
  };
  let blob, fn;
  if (fmt === "csv" || fmt === "xlsx") {
    const rows = [["Name", "Title", "Company"], ...data.leads.map(l => [l.name, l.title, l.company])];
    blob = new Blob([rows.map(r => r.join(",")).join("\n")], { type: "text/csv" });
    fn = "report." + fmt;
  } else if (fmt === "html" || fmt === "jsx") {
    blob = new Blob([document.documentElement.outerHTML], { type: "text/html" });
    fn = "report." + fmt;
  } else {
    blob = new Blob([JSON.stringify(data, null, 2)], { type: "application/json" });
    fn = "report-" + fmt + ".json";
  }
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a"); a.href = url; a.download = fn; a.click();
  toast("Exported " + fn);
}
